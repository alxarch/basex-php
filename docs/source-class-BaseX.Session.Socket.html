 58:     $total = 0;
 59:     $eof = false;

139:    * Makes sure the buffer is full.
140:    * 
141:    * @return int Number of bytes remaining in the buffer.

143:   protected function _fill()
144:   {
145:     if($this->pos >= $this->size)
146:     {
147:       // Read ahead in case of \xFF character at the end of the buffer because
148:       // \xFF could be part of an \xFF\x00 escape sequence.
149:       $chunks = array();
150:       $last = self::PAD;
151:       while($last === self::PAD)
152:       {
153:         $chunk = fread($this->socket, self::BUFFER_SIZE);
154:         $chunks[] = $chunk;
155:         $last = substr($chunk, -1);
156:       }
157:       
158:       $this->buffer = implode($chunks);
159:       
160:       $this->size = strlen($this->buffer);
161:       
162:       $this->pos = 0;
163:       
164:     }
165: 
166:     return $this->size - $this->pos;
167:   }

170:    * Send a message via the socket.
171:    * 
172:    * If $msg is a resource it will be piped to the socket.

179:   public function send($msg)
180:   {

183:       // Pipe streams into each other escaping any \x00 and \xFF characters.
184:       $total = 0;
185:       $size = 0;

194:     }

223:   public function stream($size, &$data)
224:   {
225:     $chunks = array();
226:     $total = 0;
227:     $eof = false;
228:     
229:     while($eof === false && $total < $size)
230:     {
231:       $chunksize = $this->_fill();
232:       
233:       $eof = strpos($this->buffer, self::NUL, $this->pos);
234:       
235:       if($eof !== false)
236:       {
237:         $chunksize = $eof - $this->pos;
238:       }
239:       
240:       if($chunksize + $total > $size)
241:       {
242:         $chunksize = $size - $total;
243:         $eof = false;
244:       }
245:       
246:       if($this->padded($this->pos + $chunksize))
247:       {
248:         $chunksize++;
249:         $eof = false;
250:       }
251:       
252:       $chunk = $this->readChunk($chunksize);
253:       $chunks[] = $chunk;
254:       $total += strlen($chunk);
255:     }
256:     
257:     if($eof !== false)
258:     {
259:       $this->pos++;
260:     }
261:     
262:     $data = implode($chunks);
263:     
264:     return $total;
265:   }
