299:   {
300:     $options = B::options($options);

302:     $parseropt = ('html' === $parser) ? 'HTMLOPT' : 'PARSEROPT';

313:

317:     }

321:         ->setOption('parseropt', $restore)

397:   /**
398:    * Lists database resources/collections at $path.
399:    * 
400:    * all subcollections are returned as <collection/>
401:    * @param string $path 
402:    * @return \SimpleXMLElement
403:    * 
404:    * @throws \BaseX\Error
405:    */
406:   public function getContents($path=null)
407:   {
408:     $path = (string) $path;
409:     $path = trim($path, '/');
410:     $db = $this->getName();
411:     //Frakking XQuery starts counting at 1 (+1 to trim leading '/')
412:     $start = '' === $path ? 1 : strlen($path) + 2; 
413:     $xql = <<<XQL
414: <contents>
415: {
416: for \$r in db:list-details('$db', '$path')
417:   let \$p := substring(\$r/string(), $start)
418:   let \$parts := tokenize(\$p, '/')
419:   let \$name := \$parts[1]
420:   let \$count := count(\$parts)
421:   let \$time :=  xs:dateTime(\$r/@modified-date/string())
422:   group by \$name
423:   order by -sum(\$count), \$name
424:   return if(\$count > 1) 
425:     then <collection modified-date="{max(\$time)}">{\$name}</collection>
426:     else \$r
427: }
428: </contents>
429: XQL;
430:     
431:     $data = $this->getSession()->query($xql)->execute();
432:     $xml = simplexml_load_string($data);
433:     if(false === $xml)
434:       throw new Error('Failed to get contents.');
435:     return $xml;
436:   }
